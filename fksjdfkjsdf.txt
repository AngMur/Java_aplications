package contador;

import java.awt.GridLayout;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;
import javax.swing.*;
import javax.swing.JLabel;

/**
 *
 * @author ALUMNO-17-203
 */
public class Contador extends JFrame implements Runnable, ActionListener{
    JLabel num; int i=0;  
    Thread hilo;
    JButton btnIniciar, btnDetener;
    boolean contar = true;
    /**
 *
 * @author ALUMNO-17-203
 */
    public Contador(){
        setTitle("Contador");
        setSize(200, 200);
        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        setLayout(new GridLayout(3, 1));
        num = new JLabel(i +"");
        add(num);
        hilo = new Thread(this);
        btnIniciar = new JButton("Iniciar");
        btnDetener = new JButton("Detener");
        btnIniciar.addActionListener(this);
        btnDetener.addActionListener(this);
        add(btnIniciar);
        add(btnDetener);
        
    }
    public static void main(String[] args) {
        Contador c= new Contador();
        c.setVisible(true);
        //Thread hilo = new Thread(c);
        //hilo.start();
    
    }
    public void run (){
        while (contar) {
            try {
                i=i+1;
            num.setText(i+" ");
            Thread.sleep(1000);
            } catch (Exception e) {
                System.err.println("No funciono");
            }
            
            
           
        }
    }
    
    public void actionPerformed(ActionEvent e){
        
      //  hilo = new Thread(new Contador());
        if(e.getSource().equals(btnIniciar)){
        contar = true;
        hilo.start();
        System.out.println("Trabajando");
    }
        if(e.getSource().equals(btnDetener)){
            contar = false;
            System.out.println("Detenido");
            hilo.stop();
        }
    }
    
}
